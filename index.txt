        window.loadMuro = async function() {
            const q = query(collection(db, "reflexiones"), orderBy("fecha", "desc"));
            const snap = await getDocs(q);
            const m = document.getElementById('muro'); m.innerHTML = '';
            snap.forEach(doc => {
                const data = doc.data();
                const linkTxt = data.texto.replace(/([1-3]?\s?[A-Z][a-z]+)\s(\d+):(\d+)/g, '<b style="color:var(--accent);cursor:pointer" onclick="jumpTo(\'$1\',$2,$3)">$&</b>');

            });

    <script>
        let bible = []; let dics = {h1:{}, h2:{}, g:{}}; let curLib = 0;

        window.onload = async () => {
            try {
                const res = await fetch('biblia.json');
                bible = await res.json();
                renderIndex();
                // Carga dics silenciosamente
                loadDics();
            } catch(e) { document.getElementById('index-view').innerHTML = "<h2>Error cargando libros.</h2>"; }
        };

        async function loadDics() {
            try {
                dics.h1 = await fetch('strongs_h1.json').then(r => r.json());
                dics.h2 = await fetch('strongs_h2.json').then(r => r.json());
                dics.g = await fetch('strongs_g.json').then(r => r.json());
            } catch(e) { console.warn("Diccionarios no cargados"); }
        }

        function renderIndex() {
            const at = document.getElementById('list-at'); const nt = document.getElementById('list-nt');
            at.innerHTML = ''; nt.innerHTML = '';
            bible.forEach((l, i) => {
                const btn = `<button class="btn-lib" onclick="openLib(${i})">${l.nombre}</button>`;
                if(i < 39) at.innerHTML += btn; else nt.innerHTML += btn;
            });
        }

        function openLib(i) {
            curLib = i;
            document.getElementById('index-view').classList.add('hidden');
            document.getElementById('read-view').classList.remove('hidden');
            document.getElementById('current-book').innerText = bible[i].nombre;
            const caps = document.getElementById('cap-btns'); caps.innerHTML = '';
            bible[i].capitulos.forEach((_, c) => {
                caps.innerHTML += `<button id="c-btn-${c}" onclick="showCap(${c})" style="min-width:40px; height:40px;">${c+1}</button>`;
            });
            showCap(0);
        }

        function showCap(c) {
            const textCont = document.getElementById('bible-text'); textCont.innerHTML = '';
            document.querySelectorAll('#cap-btns button').forEach(b => b.style.border = "1px solid var(--border)");
            document.getElementById('c-btn-'+c).style.border = "2px solid var(--accent)";

            bible[curLib].capitulos[c].forEach((t, v) => {
                const cleanT = t.replace(/([A-Za-zÁÉÍÓÚáéíóúñÑ]+)\s?(<[HG]\d+>)/g, (m, word, code) => {
                    const rawCode = code.replace('<','').replace('>','');
                    return `<span class="strong-link" onclick="openStrong('${rawCode}')">${word}</span>`;
                });
                textCont.innerHTML += `<div class="v-box" onclick="this.classList.toggle('v-highlight')"><b>${v+1}</b> ${cleanT}</div>`;
            });
            window.scrollTo(0,0);
        }

        function openStrong(cod) {
            let d = (curLib >= 39) ? dics.g : (cod.startsWith('H1') ? dics.h1 : dics.h2);
            let info = d[cod];
            let def = info ? (info.definition || info.strongs_def || info) : "Definición no disponible.";
            if(typeof def === 'string') def = def.replace(/<[^>]*>/g, "");
            document.getElementById('strong-data').innerHTML = `<h2>${cod}</h2><p>${def}</p>`;
            toggleModal('m-strong');
        }

        function goBack() { document.getElementById('read-view').classList.add('hidden'); document.getElementById('index-view').classList.remove('hidden'); }
        function showTab(t) {
            document.getElementById('tab-biblia').classList.add('hidden'); document.getElementById('tab-comunidad').classList.add('hidden');
            document.getElementById('nav-bib').classList.remove('active'); document.getElementById('nav-com').classList.remove('active');
            document.getElementById('tab-'+t).classList.remove('hidden'); document.getElementById('nav-'+t).classList.add('active');
            if(t === 'comunidad') window.loadMuro();
        }
        window.jumpTo = (l,c,v) => {
            const i = bible.findIndex(lib => lib.nombre.toLowerCase().includes(l.toLowerCase()));
            if(i !== -1) { showTab('biblia'); openLib(i); showCap(c-1); }
        };
        function toggleModal(id) { const m = document.getElementById(id); m.style.display = (m.style.display === 'block') ? 'none' : 'block'; }
        function updateFont(s) { document.documentElement.style.setProperty('--font', s); }
    </script>
